     public static boolean hasFriend(User owner, User friend){
         List<Friends> friendsList = dao.getFriendsByUserId(owner.getID());
         for(int i = 0; i < friendsList.size(); i++){
             if(friendsList.get(i).getFriend() == friend.getID()) {
                 return true;
             }
         }
         return false;
     }
